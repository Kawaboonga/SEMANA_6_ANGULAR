Private loadingService
Type : unknown

Servicio global encargado de manejar el estado de carga de la aplicación. Expone un signal isLoading que indica si existen procesos pendientes.

 
import { Component, computed, inject } from '@angular/core';
import { NgIf } from '@angular/common';
import { LoadingService } from '@core/services/loading.service';

@Component({
  selector: 'app-loading',
  standalone: true,
  imports: [NgIf],
  templateUrl: './loading.html',
  styleUrls: ['./loading.css'],
})
export class LoadingComponent {

  // ---------------------------------------------------------------------------
  // INYECCIÓN DE SERVICIO
  // ---------------------------------------------------------------------------

  /**
   * @description
   * Servicio global encargado de manejar el estado de carga de la aplicación.
   * Expone un signal `isLoading` que indica si existen procesos pendientes.
   */
  private loadingService = inject(LoadingService);


  // ---------------------------------------------------------------------------
  // COMPUTED: isLoading
  // ---------------------------------------------------------------------------

  /**
   * @description
   * Estado reactivo que indica si debe mostrarse el overlay de carga.
   * 
   * Este computed se recalcula automáticamente cuando el LoadingService
   * incrementa o decrementa su contador interno.
   * 
   * @return boolean — `true` cuando la aplicación está en estado de carga.
   * 
   * @example
   * <!-- loading.html -->
   * <div class="loading-backdrop" *ngIf="isLoading()">
   *    <div class="loader"></div>
   * </div>
   * 
   * @usageNotes
   * - No requiere suscripciones manuales ni ngOnDestroy.
   * - Funciona con cualquier llamada que haga `loadingService.show()` / `hide()`.
   * - Ideal para cubrir toda la ventana con un overlay mientras se cargan datos.
   */
  isLoading = computed(() => this.loadingService.isLoading());
}

<!-- =========================================================================
     LOADING OVERLAY (HTML)
     -------------------------------------------------------------------------
     Esta vista aparece únicamente cuando isLoading() === true.
     El *ngIf es reactivo gracias al computed() del componente.
     
     - Cubre toda la pantalla (definido en loading.css)
     - Muestra un spinner Bootstrap accesible
     - Bloquea interacción mientras dura la carga
   ========================================================================= -->

<div *ngIf="isLoading()" class="loading-overlay">

  <!-- Spinner centrado -->
  <div 
    class="spinner-border" 
    role="status" 
    aria-label="Cargando"
  >
    <!-- Accesibilidad: texto solo para lectores de pantalla -->
    <span class="visually-hidden">Cargando...</span>
  </div>

</div>

./loading.css 

/* ============================================================================
   LOADING OVERLAY (CSS)
   ----------------------------------------------------------------------------
   Cubre toda la pantalla y bloquea interacción mientras isLoading() === true.
   Se usa junto con loading.html y loading.component.ts.
   ---------------------------------------------------------------------------
   - position: fixed → queda por encima de todo el contenido.
   - inset: 0 → atajo para top/right/bottom/left: 0.
   - fondo semi-transparente → sensación de modal no intrusivo.
   - z-index alto (1200) → compatible con Bootstrap (modal usa 1055+).
   - display flex → centra el spinner vertical + horizontalmente.
   ============================================================================
*/

.loading-overlay {
  position: fixed;           /* Mantiene el overlay cubriendo toda la pantalla */
  inset: 0;                  /* Equivalente a top:0; right:0; bottom:0; left:0 */
  background: rgba(0,0,0,.45); /* Oscurece ligeramente sin tapar todo */
  display: flex;             /* Permite centrar el spinner */
  align-items: center;       /* Centrado vertical */
  justify-content: center;   /* Centrado horizontal */
  z-index: 1200;             /* Por encima de navbars, cards, banners, etc. */
}

.loading-overlay {
  animation: fadeIn .25s ease;
}

@keyframes fadeIn {
  from { opacity: 0; }
  to   { opacity: 1; }
}

.loading-overlay {
  backdrop-filter: blur(4px);
}

.spinner-border {
  width: 3rem;
  height: 3rem;
  color: var(--bs-primary);
}
